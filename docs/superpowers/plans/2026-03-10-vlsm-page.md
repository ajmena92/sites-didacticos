  // ═══════════════════════════════════════
  //  UTILIDADES (de subnetting-practica.html)
  // ═══════════════════════════════════════
  function prefixToMask(p) {
    return Array.from({length:4}, (_,i) => {
      const b = Math.min(8, Math.max(0, p - i*8));
      return b===8 ? 255 : b===0 ? 0 : 256 - Math.pow(2, 8-b);
    });
  }
  function wildcard(m) { return m.map(x=>255-x); }
  function netAddr(ip, m) { return ip.map((b,i)=>b&m[i]); }
  function broadcast(net, wc) { return net.map((b,i)=>b+wc[i]); }
  function addOne(o) {
    const n=(((o[0]<<24)|(o[1]<<16)|(o[2]<<8)|o[3])>>>0)+1;
    return [(n>>>24)&255,(n>>>16)&255,(n>>>8)&255,n&255];
  }
  function subOne(o) {
    const n=(((o[0]<<24)|(o[1]<<16)|(o[2]<<8)|o[3])>>>0)-1;
    return [(n>>>24)&255,(n>>>16)&255,(n>>>8)&255,n&255];
  }
  function str(o) { return o.join('.'); }
  function normIP(s) {
    if (!s) return null;
    const parts = s.trim().replace(/\s+/g,'.').replace(/,/g,'.').split('.');
    if (parts.length!==4) return null;
    const n = parts.map(p=>parseInt(p,10));
    if (n.some(x=>isNaN(x)||x<0||x>255)) return null;
    return n.join('.');
  }
  function normPfx(s) {
    const t = s.trim().replace(/^\//, '');
    const n = parseInt(t, 10);
    return (isNaN(n)||n<0||n>32) ? null : n;
  }

  // ═══════════════════════════════════════
  //  CONFIGURACIÓN DE NIVELES
  // ═══════════════════════════════════════
  const LEVELS = {
    basico: {
      label: 'Básico', cls: 'basico',
      // hosts: 3 subredes, prefijos entre /24 y /27 (hosts 30-254)
      hostRanges: [[30,62],[62,126],[126,254]],
      bases: () => {
        const x = ri(1,254);
        return { ip:[192,168,x,0], prefix:24 };
      }
    },
    intermedio: {
      label: 'Intermedio', cls: 'intermedio',
      // hosts: 5 subredes, prefijos /22-/28 (hosts 6-1022)
      hostRanges: [[6,14],[14,30],[30,62],[62,126],[500,1022]],
      bases: () => {
        const x = ri(16,31), y = ri(0,15);
        return { ip:[172,x,y*16,0], prefix:20 };
      }
    },
    avanzado: {
      label: 'Avanzado', cls: 'avanzado',
      // hosts: 6 subredes LAN /19-/28, + 4 WAN /30
      hostRanges: [[2000,4000],[500,1022],[200,500],[100,200],[30,62],[6,14]],
      bases: () => {
        const x = ri(1,200);
        return { ip:[10,x,0,0], prefix:16 };
      }
    }
  };

  function ri(min,max){ return min + Math.floor(Math.random()*(max-min+1)); }

  // ═══════════════════════════════════════
  //  ESTADO GLOBAL
  // ═══════════════════════════════════════
  const V = {
    subnets:[],      // [{hosts, prefix, mask, net, first, last, bc, wan}] ordenadas mayor→menor
    hostsGiven:[],   // orden aleatorio para mostrar en chips
    activeRow:0,
    done:false
  };

  // ═══════════════════════════════════════
  //  GENERADOR DE EJERCICIO
  // ═══════════════════════════════════════
  function newExercise() {
    const lvl = document.getElementById('lvlSelect').value;
    const cfg = LEVELS[lvl];
    const badge = document.getElementById('lvlBadge');
    badge.textContent = cfg.label;
    badge.className = 'level-badge ' + cfg.cls;

    const base = cfg.bases();
    let cur = [...base.ip];

    // Generar hosts LAN
    const lanHosts = cfg.hostRanges.map(([lo,hi]) => ri(lo,hi));
    // Ordenar de mayor a menor para VLSM
    lanHosts.sort((a,b)=>b-a);

    // Generar subredes WAN (2 hosts /30) para avanzado
    const wanHosts = (lvl === 'avanzado') ? [2,2,2,2] : [];
    const allHostsSorted = [...lanHosts, ...wanHosts]; // WAN siempre al final

    // Calcular subredes VLSM
    V.subnets = allHostsSorted.map((h, idx) => {
      // Prefijo mínimo: 32 - ceil(log2(h+2))
      const bits = Math.ceil(Math.log2(h + 2));
      const p = 32 - bits;
      const m = prefixToMask(p);
      const wc = wildcard(m);
      const net = netAddr(cur, m);
      const bc = broadcast(net, wc);
      cur = addOne(bc);
      return {
        hosts: h,
        prefix: p,
        mask: m,
        net: [...net],
        first: addOne(net),
        last: subOne(bc),
        bc: [...bc],
        wan: idx >= lanHosts.length
      };
    });

    // Hosts para el pool — en orden aleatorio
    const allForPool = [...lanHosts, ...wanHosts];
    V.hostsGiven = [...allForPool].sort(() => Math.random() - .5);

    V.activeRow = 0;
    V.done = false;
    document.getElementById('doneBanner').style.display = 'none';

    renderPool();
    renderTable();
  }

  // ═══════════════════════════════════════
  //  RENDER POOL
  // ═══════════════════════════════════════
  function renderPool() {
    const container = document.getElementById('poolChips');
    container.innerHTML = '';
    // El siguiente a asignar es V.subnets[V.activeRow].hosts
    const nextHosts = V.done ? null : V.subnets[V.activeRow]?.hosts;
    V.hostsGiven.forEach((h, i) => {
      const used = V.subnets.slice(0, V.activeRow).some(s => s.hosts === h && !s._chipUsed)
        ? (() => { /* mark consumed */ })()
        : false;
      const chip = document.createElement('span');
      chip.className = 'chip';
      chip.id = 'chip-' + i;
      chip.textContent = h;
      container.appendChild(chip);
    });
    // Mark used chips (consumed subnets, largest first)
    const consumed = V.subnets.slice(0, V.activeRow).map(s=>s.hosts);
    const remaining = [...V.hostsGiven];
    consumed.forEach(h => {
      const idx = remaining.indexOf(h);
      if (idx !== -1) {
        const chip = document.getElementById('chip-' + V.hostsGiven.indexOf(h));
        if (chip && !chip.classList.contains('used')) chip.classList.add('used');
        remaining.splice(idx, 1);
      }
    });
    // Highlight next chip
    if (nextHosts !== null && nextHosts !== undefined) {
      const nextIdx = V.hostsGiven.findIndex((h,i) => {
        const chip = document.getElementById('chip-' + i);
        return h === nextHosts && chip && !chip.classList.contains('used');
      });
      if (nextIdx !== -1) {
        document.getElementById('chip-' + nextIdx)?.classList.add('next');
      }
    }
  }

  // ═══════════════════════════════════════
  //  RENDER TABLE
  // ═══════════════════════════════════════
  function renderTable() {
    const tbody = document.getElementById('vlsmBody');
    tbody.innerHTML = '';
    V.subnets.forEach((s, i) => {
      const tr = document.createElement('tr');
      const state = i < V.activeRow ? 'row-done' : i === V.activeRow ? 'row-active' : 'row-locked';
      tr.className = state + (s.wan ? ' row-wan' : '');
      tr.id = 'row-' + i;

      if (i < V.activeRow) {
        // Fila completada: mostrar valores correctos
        tr.innerHTML = `
          <td class="rnum">${i+1}</td>
          <td><input class="ti ti-h ok" value="${s.hosts}" readonly /></td>
          <td><input class="ti ti-ip ok" value="${str(s.net)}" readonly /></td>
          <td><input class="ti ti-px ok" value="/${s.prefix}" readonly /></td>
          <td><input class="ti ti-mk ok" value="${str(s.mask)}" readonly /></td>
          <td><input class="ti ti-ip ok" value="${str(s.first)}" readonly /></td>
          <td><input class="ti ti-ip ok" value="${str(s.last)}" readonly /></td>
          <td><input class="ti ti-ip ok" value="${str(s.bc)}" readonly /></td>
          <td><span class="tag-ok">✓</span></td>`;
      } else if (i === V.activeRow) {
        // Fila activa: inputs editables
        tr.innerHTML = `
          <td class="rnum">${i+1}</td>

  function verify(i) {
    const s = V.subnets[i];
    const get = id => document.getElementById(id)?.value ?? '';

    const inH   = parseInt(get('in-h-'   + i), 10);
    const inNet  = normIP(get('in-net-' + i));
    const inPx   = normPfx(get('in-px-' + i));
    const inMk   = normIP(get('in-mk-'  + i));
    const inF    = normIP(get('in-f-'   + i));
    const inL    = normIP(get('in-l-'   + i));
    const inBc   = normIP(get('in-bc-'  + i));

    const errors = [];

    // Hosts: debe ser el mayor pendiente (= s.hosts)
    if (inH !== s.hosts) {
      errors.push(`Hosts: debes asignar el mayor pendiente (${s.hosts})`);
    }
    // Dir. de Red
    if (inNet !== str(s.net)) {
      const expected = i === 0
        ? str(s.net) + ' (red base del ejercicio)'
        : str(s.net) + ' (Broadcast anterior + 1)';
      errors.push(`Dir. de Red: debe ser ${expected}`);
    }
    // Prefijo
    if (inPx !== s.prefix) {
      errors.push(`Prefijo: para ${s.hosts} hosts el mínimo es /${s.prefix}  →  2^${32-s.prefix}−2 = ${Math.pow(2,32-s.prefix)-2} hosts útiles`);
    }
    // Máscara
    if (inMk !== str(s.mask)) {
      errors.push(`Máscara: /${s.prefix} corresponde a ${str(s.mask)}`);
    }
    // Primer host
    if (inF !== str(s.first)) {
      errors.push(`Primer host: ${str(s.first)}  (Dir. de Red + 1)`);
    }
    // Último host
    if (inL !== str(s.last)) {
      errors.push(`Último host: ${str(s.last)}  (Broadcast − 1)`);
    }
    // Broadcast
    if (inBc !== str(s.bc)) {
      errors.push(`Broadcast: ${str(s.bc)}  (Dir. de Red + ${Math.pow(2,32-s.prefix)} − 1)`);
    }

    // Eliminar feedback previo
    const prevFb = document.getElementById('fb-' + i);
    if (prevFb) prevFb.remove();

    if (errors.length === 0) {
      // ✓ Correcto
      V.activeRow++;
      if (V.activeRow >= V.subnets.length) {
        V.done = true;
        document.getElementById('doneBanner').style.display = 'block';
      }
      renderPool();
      renderTable();
    } else {
      // ✗ Error: mostrar fila de feedback + shake
      const activeRow = document.getElementById('row-' + i);
      activeRow?.classList.add('shake');
      setTimeout(() => activeRow?.classList.remove('shake'), 350);

      const fbTr = document.createElement('tr');
      fbTr.id = 'fb-' + i;
      fbTr.className = 'fb-row';
      fbTr.innerHTML = `<td colspan="9"><div class="fb-msg">⚠ ${errors.join('<br>⚠ ')}</div></td>`;
      activeRow?.insertAdjacentElement('afterend', fbTr);
    }
  }

> **Note:** The chip "used" tracking in Task 6 has a subtle issue — the pool render uses `hostsGiven` index to find chips, but consumed hosts may appear multiple times in the pool (especially for WAN /30 with 4×"2" chips). This task adds precise per-chip tracking.

- [ ] **Step 1: Replace `renderPool` with index-based tracking**

  The corrected approach: track consumed counts per host value separately from display indices.

  ```javascript
  function renderPool() {
    const container = document.getElementById('poolChips');
    container.innerHTML = '';

    // Count how many of each host value have been consumed (completed rows)
    const consumedCount = {};
    V.subnets.slice(0, V.activeRow).forEach(s => {
      consumedCount[s.hosts] = (consumedCount[s.hosts] || 0) + 1;
    });

    // Working copy of consumed counts for marking chips
    const usedLeft = {...consumedCount};

    const nextHosts = V.done ? null : V.subnets[V.activeRow]?.hosts;
    let nextMarked = false;

    V.hostsGiven.forEach((h, i) => {
      const chip = document.createElement('span');
      chip.className = 'chip';
      chip.id = 'chip-' + i;
      chip.textContent = h;

      if (usedLeft[h] > 0) {
        chip.classList.add('used');
        usedLeft[h]--;
      } else if (!nextMarked && h === nextHosts) {
        chip.classList.add('next');
        nextMarked = true;
      }
      container.appendChild(chip);
    });
  }
  ```
